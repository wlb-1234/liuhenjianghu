// 1. 安装SDK npm install @regions/api-client // 2. 初始化客户端 import { RegionsClient } from '@regions/api-client'; const client = new RegionsClient({ apiKey: 'YOUR_API_KEY', baseUrl: 'https://your-api.railway.app' }); // 3. 查询省级列表 const provinces = await client.provinces(); // 4. 查询城市下级 const cities = await client.cities('44'); // 广东省 // 5. 反向地理编码 const result = await client.reverse(39.9042, 116.4074);

接口说明示例
GET /provinces省级列表/api/v1/regions/provinces
GET /cities/:code城市列表/api/v1/regions/cities/44

GET /reverse反向地理编码/api/v1/geo/reverse?lat=39.9&lng=116.4
GET /distance距离计算/api/v1/geo/distance?fromLat=39.9&fromLng=116.4&toLat=31.2&toLng=121.5
